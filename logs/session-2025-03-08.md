### Usage Example

```kotlin
// Initialize provider
val offlineProvider = OfflineLLMProvider(context, llamaBridge)
offlineProvider.initialize()

// Create request with memory context
val request = CompletionRequest(
    messages = listOf(ChatMessage.user("What's my mom's name?")),
    systemPrompt = "You are a helpful assistant.",
    memoryContext = "[MEMORY]\nUser: Kali\nPeople: Priya (mother)\n[/MEMORY]",
    maxTokens = 256,
    stream = true
)

// Stream response
offlineProvider.stream(request).collect { response ->
    when {
        response.isStreaming -> print(response.text)
        response.isComplete -> println("\nDone! (${response.latencyMs}ms)")
        response.isError -> println("Error: ${response.error}")
    }
}

// Or simple blocking call
val answer = offlineProvider.completeSimple(
    userMessage = "Hello!",
    memoryContext = memoryManager?.buildContextBlock()
)
```
